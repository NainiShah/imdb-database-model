--NEWS VIEW
CREATE VIEW NEWS_VIEW AS
SELECT NEWS.TITLE, NEWS.PUBLISH_DATE, NEWS.DETAILS
, CASE
WHEN NEWS.TV_SHOW_ID IS NOT NULL THEN 'TV SHOW' 
WHEN NEWS.MOVIE_ID IS NOT NULL THEN 'MOVIE' 
WHEN NEWS.AWARD_ID IS NOT NULL THEN 'AWARD' 
WHEN NEWS.CAST_ID IS NOT NULL THEN 'CAST' 
WHEN NEWS.THEATRE_ID IS NOT NULL THEN 'THEATRE' 
WHEN NEWS.SHOWTIME_ID IS NOT NULL THEN 'SHOWTIME' 
END AS REVIEW_TYPE
FROM NEWS
